158 void OdomTracker::popPath(int popCount, bool keepPreviousPath)
159 {
160  RCLCPP_INFO(getLogger(), "odom_tracker m_mutex acquire");
161  std::lock_guard<std::mutex> lock(m_mutex_);
162 
163  RCLCPP_INFO(getLogger(), "POP PATH ENTRY");
164  this->logStateString();
165 
166  if (!keepPreviousPath)
167  {
168  baseTrajectory_.poses.clear();
169  }
170 
171  while (popCount > 0 && !pathStack_.empty())
172  {
173  auto & stacked = pathStack_.back().poses;
174  baseTrajectory_.poses.insert(baseTrajectory_.poses.begin(), stacked.begin(), stacked.end());
175  pathStack_.pop_back();
176  popCount--;
177 
178  RCLCPP_INFO(getLogger(), "POP PATH Iteration ");
179  this->logStateString();
180  }
181 
182  RCLCPP_INFO(getLogger(), "POP PATH EXITING");
183  this->logStateString();
184  RCLCPP_INFO(getLogger(), "odom_tracker m_mutex release");
185  this->updateAggregatedStackPath();
186 }

285 bool OdomTracker::updateClearPath(const nav_msgs::msg::Odometry & odom)
286 {
287  // we initially accept any message if the queue is empty
289  geometry_msgs::msg::PoseStamped base_pose;
290 
291  base_pose.pose = odom.pose.pose;
292  base_pose.header = odom.header;
293  baseTrajectory_.header = odom.header;
294 
295  bool acceptBackward = false;
296  bool clearingError = false;
297  bool finished = false;
298 
299  while (!finished)
300  {
301  if (
302  baseTrajectory_.poses.size() <=
303  1) // we at least keep always the first point of the forward path when clearing
304  // (this is important for backwards planner replanning and not losing the
305  // last goal)
306  {
307  acceptBackward = false;
308  finished = true;
309  }
310  else
311  {
312  auto & carrotPose = baseTrajectory_.poses.back().pose;
313  auto & carrotPoint = carrotPose.position;
314  double carrotAngle = tf2::getYaw(carrotPose.orientation);
315 
316  auto & currePose = base_pose.pose;
317  auto & currePoint = currePose.position;
318  double currentAngle = tf2::getYaw(currePose.orientation);
319 
320  double lastpointdist = p2pDistance(carrotPoint, currePoint);
321  double goalAngleOffset = fabs(angles::shortest_angular_distance(carrotAngle, currentAngle));
322 
323  acceptBackward = !baseTrajectory_.poses.empty() &&
324  lastpointdist < clearPointDistanceThreshold_ &&
325  goalAngleOffset < clearAngularDistanceThreshold_;
326 
327  clearingError = lastpointdist > 2 * clearPointDistanceThreshold_;
328  RCLCPP_DEBUG_STREAM(
329  getLogger(), "[OdomTracker] clearing (accepted: " << acceptBackward
330  << ") linerr: " << lastpointdist
331  << ", anglerr: " << goalAngleOffset);
332  }
333 
334  // RCLCPP_INFO(getLogger(),"Backwards, last distance: %lf < %lf accept: %d", dist,
335  // minPointDistanceBackwardThresh_, acceptBackward);
336  if (
337  acceptBackward &&
338  baseTrajectory_.poses.size() > 1) /* we always leave at least one item, specially interesting
339  for the backward local planner reach the backwards goal
340  with enough precision*/
341  {
342  baseTrajectory_.poses.pop_back();
343  }
344  else if (clearingError)
345  {
346  finished = true;
347  RCLCPP_WARN(getLogger(), "[OdomTracker] Incorrect odom clearing motion.");
348  }
349  else
350  {
351  finished = true;
353  }
354  }
355 
356  return acceptBackward;
357 }
363 bool OdomTracker::updateRecordPath(const nav_msgs::msg::Odometry & odom)
364 {
366  geometry_msgs::msg::PoseStamped base_pose;
367 
368  base_pose.pose = odom.pose.pose;
369  base_pose.header = odom.header;
370  baseTrajectory_.header = odom.header;
371 
372  bool enqueueOdomMessage = false;
373 
374  double dist = -1;
375  if (baseTrajectory_.poses.empty())
376  {
377  enqueueOdomMessage = true;
378  }
379  else
380  {
381  const auto & prevPose = baseTrajectory_.poses.back().pose;
382  const geometry_msgs::msg::Point & prevPoint = prevPose.position;
383  double prevAngle = tf2::getYaw(prevPose.orientation);
384 
385  const geometry_msgs::msg::Point & currePoint = base_pose.pose.position;
386  double currentAngle = tf2::getYaw(base_pose.pose.orientation);
387 
388  dist = p2pDistance(prevPoint, currePoint);
389  double goalAngleOffset = fabs(angles::shortest_angular_distance(prevAngle, currentAngle));
390 
391  // RCLCPP_WARN(getLogger(),"dist %lf vs min %lf", dist, recordPointDistanceThreshold_);
392 
393  if (dist > recordPointDistanceThreshold_ || goalAngleOffset > recordAngularDistanceThreshold_)
394  {
395  enqueueOdomMessage = true;
396  }
397  else
398  {
399  // RCLCPP_WARN(getLogger(),"skip odom, dist: %lf", dist);
400  enqueueOdomMessage = false;
401  }
402  }
403 
404  if (enqueueOdomMessage)
405  {
406  baseTrajectory_.poses.push_back(base_pose);
407  }
408 
409  return enqueueOdomMessage;
410 }
