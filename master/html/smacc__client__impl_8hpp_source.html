67 SmaccComponentType * ISmaccClient::createNamedComponent(std::string name, TArgs... targs)
68 {
69  ComponentKey componentkey(&typeid(SmaccComponentType), name);
70 
71  std::shared_ptr<SmaccComponentType> ret;
72 
73  auto it = this->components_.find(componentkey);
74 
75  if (it == this->components_.end())
76  {
77  auto tname = demangledTypeName<SmaccComponentType>();
78  RCLCPP_INFO(
79  getLogger(),
80  "Creating a new component of type %s smacc component is required by client '%s'. Creating a "
81  "new instance %s",
82  this->getName().c_str(), demangledTypeName<SmaccComponentType>().c_str(), tname.c_str());
83 
84  ret = std::shared_ptr<SmaccComponentType>(new SmaccComponentType(targs...));
85  ret->setStateMachine(this->getStateMachine());
86  ret->owner_ = this;
87  ret->initialize(this);
88 
89  this->components_[componentkey] =
90  ret; //std::dynamic_pointer_cast<smacc2::ISmaccComponent>(ret);
91  RCLCPP_DEBUG(getLogger(), "%s resource is required. Done.", tname.c_str());
92  }
93  else
94  {
95  RCLCPP_INFO(
96  getLogger(), "%s resource is required. Found resource in cache.",
97  demangledTypeName<SmaccComponentType>().c_str());
98  ret = dynamic_pointer_cast<SmaccComponentType>(it->second);
99  }
100 
101  ret->template onOrthogonalAllocation<TOrthogonal, TClient>();
102 
103  return ret.get();
104 }
