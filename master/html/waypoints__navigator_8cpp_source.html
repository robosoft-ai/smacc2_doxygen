77 void WaypointNavigator::sendNextGoal()
78 {
79  if (currentWaypoint_ >= 0 && currentWaypoint_ < (int)waypoints_.size())
80  {
81  auto & next = waypoints_[currentWaypoint_];
82 
83  ClNav2Z::Goal goal;
84  auto p = client_->getComponent<cl_nav2z::Pose>();
85  auto pose = p->toPoseMsg();
86 
87  // configuring goal
88  goal.pose.header.frame_id = p->getReferenceFrame();
89  goal.pose.header.stamp = getNode()->now();
90  goal.pose.pose = next;
91 
92  RCLCPP_WARN(getLogger(), "[WaypointsNavigator] Configuring default planners");
93  auto plannerSwitcher = client_->getComponent<PlannerSwitcher>();
94  plannerSwitcher->setDefaultPlanners();
95 
96  RCLCPP_WARN(getLogger(), "[WaypointsNavigator] Configuring default goal planner");
97  auto goalCheckerSwitcher = client_->getComponent<GoalCheckerSwitcher>();
98  goalCheckerSwitcher->setGoalCheckerId("goal_checker");
99 
100  // publish stuff
101  // rclcpp::sleep_for(5s);
102 
103  RCLCPP_INFO(getLogger(), "[WaypointsNavigator] Getting odom tracker");
104  auto odomTracker = client_->getComponent<cl_nav2z::odom_tracker::OdomTracker>();
105  if (odomTracker != nullptr)
106  {
107  RCLCPP_INFO(getLogger(), "[WaypointsNavigator] Storing path in odom tracker");
108 
109  auto pathname = this->owner_->getStateMachine()->getCurrentState()->getName() + " - "+ getName();
110  odomTracker->pushPath(pathname);
111  odomTracker->setStartPoint(pose);
112  odomTracker->setWorkingMode(cl_nav2z::odom_tracker::WorkingMode::RECORD_PATH);
113  }
114 
115  // SEND GOAL
116  if (!succeddedNav2ZClientConnection_.connected())
117  {
118  this->succeddedNav2ZClientConnection_ =
119  client_->onSucceeded(&WaypointNavigator::onGoalReached, this);
120  this->cancelledNav2ZClientConnection_ =
121  client_->onAborted(&WaypointNavigator::onGoalCancelled, this);
122  this->abortedNav2ZClientConnection_ =
123  client_->onCancelled(&WaypointNavigator::onGoalAborted, this);
124  }
125 
126  client_->sendGoal(goal);
127  }
128  else
129  {
130  RCLCPP_WARN(
131  getLogger(),
132  "[WaypointsNavigator] All waypoints were consumed. There is no more waypoints available.");
133  }
134 }

149 void WaypointNavigator::setWaypoints(const std::vector<Pose2D> & waypoints)
150 {
151  waypoints_.clear();
152  waypointsNames_.clear();
153  int i = 0;
154  for (auto & p : waypoints)
155  {
156  geometry_msgs::msg::Pose pose;
157  pose.position.x = p.x_;
158  pose.position.y = p.y_;
159  pose.position.z = 0.0;
160  tf2::Quaternion q;
161  q.setRPY(0, 0, p.yaw_);
162  pose.orientation = tf2::toMsg(q);
163 
164  waypoints_.push_back(pose);
165  waypointsNames_.push_back(std::to_string(i++));
166  }
167 }

185 void WaypointNavigator::loadWayPointsFromFile(std::string filepath)
186 {
187  RCLCPP_INFO_STREAM(getLogger(), "[WaypointNavigator] Loading file:" << filepath);
188  this->waypoints_.clear();
189  std::ifstream ifs(filepath.c_str(), std::ifstream::in);
190  if (ifs.good() == false)
191  {
192  throw std::string("Waypoints file not found");
193  }
194 
195  try
196  {
197 #ifdef HAVE_NEW_YAMLCPP
198  YAML::Node node = YAML::Load(ifs);
199 #else
200  YAML::Parser parser(ifs);
201  parser.GetNextDocument(node);
202 #endif
203 
204 #ifdef HAVE_NEW_YAMLCPP
205  const YAML::Node & wp_node_tmp = node["waypoints"];
206  const YAML::Node * wp_node = wp_node_tmp ? &wp_node_tmp : NULL;
207 #else
208  const YAML::Node * wp_node = node.FindValue("waypoints");
209 #endif
210 
211  if (wp_node != NULL)
212  {
213  for (unsigned int i = 0; i < wp_node->size(); ++i)
214  {
215  // Parse waypoint entries on YAML
216  geometry_msgs::msg::Pose wp;
217 
218  try
219  {
220  // (*wp_node)[i]["name"] >> wp.name;
221  // (*wp_node)[i]["frame_id"] >> wp.header.frame_id;
222  wp.position.x = (*wp_node)[i]["position"]["x"].as<double>();
223  wp.position.y = (*wp_node)[i]["position"]["y"].as<double>();
224  wp.position.z = (*wp_node)[i]["position"]["z"].as<double>();
225  wp.orientation.x = (*wp_node)[i]["orientation"]["x"].as<double>();
226  wp.orientation.y = (*wp_node)[i]["orientation"]["y"].as<double>();
227  wp.orientation.z = (*wp_node)[i]["orientation"]["z"].as<double>();
228  wp.orientation.w = (*wp_node)[i]["orientation"]["w"].as<double>();
229 
230  this->waypoints_.push_back(wp);
231  }
232  catch (...)
233  {
234  RCLCPP_ERROR(getLogger(), "parsing waypoint file, syntax error in point %d", i);
235  }
236  }
237  RCLCPP_INFO_STREAM(getLogger(), "Parsed " << this->waypoints_.size() << " waypoints.");
238  }
239  else
240  {
241  RCLCPP_WARN_STREAM(getLogger(), "Couldn't find any waypoints in the provided yaml file.");
242  }
243  }
244  catch (const YAML::ParserException & ex)
245  {
246  RCLCPP_ERROR_STREAM(
247  getLogger(), "Error loading the Waypoints YAML file. Incorrect syntax: " << ex.what());
248  }
