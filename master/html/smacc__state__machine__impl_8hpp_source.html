172 template <typename EventType>
173 void ISmaccStateMachine::postEvent(EventType * ev, EventLifeTime evlifetime)
174 {
175  std::lock_guard<std::recursive_mutex> guard(eventQueueMutex_);
176 
177 #define eventtypename demangleSymbol<EventType>().c_str()
178 
179  TRACEPOINT(smacc2_event, eventtypename);
180 
181  if (
182  evlifetime == EventLifeTime::CURRENT_STATE &&
183  (stateMachineCurrentAction == StateMachineInternalAction::STATE_EXITING ||
184  stateMachineCurrentAction == StateMachineInternalAction::TRANSITIONING))
185  {
186  RCLCPP_WARN_STREAM(
187  getLogger(),
188  "CURRENT STATE SCOPED EVENT SKIPPED, state is exiting/transitioning " << eventtypename);
189  return;
190  // in this case we may lose/skip events, if this is not right for some cases we should create a
191  // queue to lock the events during the transitions. This issues appeared when a client
192  // asyncbehavior was posting an event meanwhile we were doing the transition, but the main
193  // thread was waiting for its correct finalization (with thread.join)
194  }
195 
196  // when a postting event is requested by any component, client, or client behavior
197  // we reach this place. Now, we propagate the events to all the state state reactors to generate
198  // some more events
199 
200  RCLCPP_DEBUG_STREAM(getLogger(), "[PostEvent entry point] " << eventtypename);
201  auto currentstate = currentState_;
202  if (currentstate != nullptr)
203  {
204  propagateEventToStateReactors(currentstate, ev);
205  }
206 
207  this->signalDetector_->postEvent(ev);
208 }

217 template <typename T>
218 bool ISmaccStateMachine::getGlobalSMData(std::string name, T & ret)
219 {
220  std::lock_guard<std::recursive_mutex> lock(m_mutex_);
221  // RCLCPP_WARN(getLogger(),"get SM Data lock acquire");
222  bool success = false;
223 
224  if (!globalData_.count(name))
225  {
226  // RCLCPP_WARN(getLogger(),"get SM Data - data do not exist");
227  success = false;
228  }
229  else
230  {
231  // RCLCPP_WARN(getLogger(),"get SM DAta -data exist. accessing");
232  try
233  {
234  auto & v = globalData_[name];
235 
236  // RCLCPP_WARN(getLogger(),"get SM DAta -data exist. any cast");
237  ret = boost::any_cast<T>(v.second);
238  success = true;
239  // RCLCPP_WARN(getLogger(),"get SM DAta -data exist. success");
240  }
241  catch (boost::bad_any_cast & ex)
242  {
243  RCLCPP_ERROR(getLogger(), "bad any cast: %s", ex.what());
244  success = false;
245  }
246  }
247 
248  // RCLCPP_WARN(getLogger(),"get SM Data lock release");
249  return success;
250 }

272 template <typename StateField, typename BehaviorType>
273 void ISmaccStateMachine::mapBehavior()
274 {
275  std::string stateFieldName = demangleSymbol(typeid(StateField).name());
276  std::string behaviorType = demangleSymbol(typeid(BehaviorType).name());
277  RCLCPP_INFO(
278  getLogger(), "Mapping state field '%s' to stateReactor '%s'", stateFieldName.c_str(),
279  behaviorType.c_str());
280  SmaccClientBehavior * globalreference;
281  if (!this->getGlobalSMData(stateFieldName, globalreference))
282  {
283  // Using the requires component approach, we force a unique existence
284  // of this component
285  BehaviorType * behavior;
286  this->requiresComponent(behavior);
287  globalreference = dynamic_cast<ISmaccClientBehavior *>(behavior);
288 
289  this->setGlobalSMData(stateFieldName, globalreference);
290  }

350 template <typename TSmaccSignal, typename TMemberFunctionPrototype, typename TSmaccObjectType>
351 boost::signals2::connection ISmaccStateMachine::createSignalConnection(
352  TSmaccSignal & signal, TMemberFunctionPrototype callback, TSmaccObjectType * object)
353 {
354  static_assert(
355  std::is_base_of<ISmaccState, TSmaccObjectType>::value ||
356  std::is_base_of<ISmaccClient, TSmaccObjectType>::value ||
357  std::is_base_of<ISmaccClientBehavior, TSmaccObjectType>::value ||
358  std::is_base_of<StateReactor, TSmaccObjectType>::value ||
359  std::is_base_of<ISmaccComponent, TSmaccObjectType>::value,
360  "Only are accepted smacc types as subscribers for smacc signals");
361 
362  typedef decltype(callback) ft;
363  Bind<boost::function_types::function_arity<ft>::value> binder;
364  boost::signals2::connection connection = binder.bindaux(signal, callback, object);
365 
366  // long life-time objects
367  if (
368  std::is_base_of<ISmaccComponent, TSmaccObjectType>::value ||
369  std::is_base_of<ISmaccClient, TSmaccObjectType>::value ||
370  std::is_base_of<ISmaccOrthogonal, TSmaccObjectType>::value ||
371  std::is_base_of<ISmaccStateMachine, TSmaccObjectType>::value)
372  {
373  }
374  else if (
375  std::is_base_of<ISmaccState, TSmaccObjectType>::value ||
376  std::is_base_of<StateReactor, TSmaccObjectType>::value ||
377  std::is_base_of<ISmaccClientBehavior, TSmaccObjectType>::value)
378  {
379  RCLCPP_INFO(
380  getLogger(),
381  "[StateMachine] life-time constrained smacc signal subscription created. Subscriber is %s",
382  demangledTypeName<TSmaccObjectType>().c_str());
383  stateCallbackConnections.push_back(connection);
384  }
385  else // state life-time objects
386  {
387  RCLCPP_WARN(
388  getLogger(),
389  "[StateMachine] connecting signal to an unknown object with life-time unknown "
390  "behavior. It might provoke "
391  "an exception if the object is destroyed during the execution.");
392  }
393 
394  return connection;
395 }

415 {

417  getLogger(), "[%s] State OnEntry code finished",
418  demangleSymbol(typeid(StateType).name()).c_str());

420  for (auto pair : this->orthogonals_)
421  {
422  RCLCPP_DEBUG(getLogger(), "ortho onentry: %s", pair.second->getName().c_str());

426  orthogonal->onEntry();

448  getLogger(),

452  }
453  }
454 
455  for (auto & eg : this->currentState_->getEventGenerators())
456  {
