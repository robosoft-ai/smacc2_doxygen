39 #define THIS_STATE_NAME ((demangleSymbol(typeid(*this).name()).c_str()))
40 template <typename TOrthogonal, typename TBehavior, typename... Args>
41 std::shared_ptr<TBehavior> ISmaccState::configure(Args &&... args)
42 {
43  std::string orthogonalkey = demangledTypeName<TOrthogonal>();
44  RCLCPP_INFO(
45  getLogger(), "[%s] Configuring orthogonal: %s", THIS_STATE_NAME, orthogonalkey.c_str());
46 
47  TOrthogonal * orthogonal = this->getOrthogonal<TOrthogonal>();
48  if (orthogonal != nullptr)
49  {
50  auto clientBehavior = std::shared_ptr<TBehavior>(new TBehavior(args...));
51  clientBehavior->currentState = this;
52  orthogonal->addClientBehavior(clientBehavior);
53  clientBehavior->template onOrthogonalAllocation<TOrthogonal, TBehavior>();
54  return clientBehavior;
55  }
56  else
57  {
58  RCLCPP_ERROR(
59  getLogger(), "[%s] Skipping client behavior creation in orthogonal [%s]. It does not exist.",
60  THIS_STATE_NAME, orthogonalkey.c_str());
61  return nullptr;
62  }
63 }

73 template <typename SmaccClientType>
74 void ISmaccState::requiresClient(SmaccClientType *& storage)
75 {
76  const char * sname = (demangleSymbol(typeid(*this).name()).c_str());
77  storage = nullptr;
78  auto & orthogonals = this->getStateMachine().getOrthogonals();
79  for (auto & ortho : orthogonals)
80  {
81  ortho.second->requiresClient(storage);
82  if (storage != nullptr) return;
83  }
84 
85  RCLCPP_ERROR(
86  getLogger(),
87  "[%s] Client of type '%s' not found in any orthogonal of the current state machine. This may "
88  "produce a segmentation fault if the returned reference is used.",
89  sname, demangleSymbol<SmaccClientType>().c_str());
90 }
