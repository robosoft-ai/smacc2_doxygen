139  marker.color.g = g;
140  marker.color.b = b;
141 
142  geometry_msgs::msg::Point start, end;
143  start.x = pose.position.x;
144  start.y = pose.position.y;
145 
146  end.x = pose.position.x + 0.5 * cos(phi);
147  end.y = pose.position.y + 0.5 * sin(phi);
148 
149  marker.points.push_back(start);
150  marker.points.push_back(end);
151 
152  visualization_msgs::msg::MarkerArray ma;
153  ma.markers.push_back(marker);
154 
155  markersPub_->publish(ma);

163 void BackwardGlobalPlanner::createDefaultBackwardPath(
164  const geometry_msgs::msg::PoseStamped & start, const geometry_msgs::msg::PoseStamped & goal,
165  std::vector<geometry_msgs::msg::PoseStamped> & plan)
166 {
167  auto q = start.pose.orientation;
168 
169  geometry_msgs::msg::PoseStamped pose;
170  pose = start;
171 
172  double dx = start.pose.position.x - goal.pose.position.x;
173  double dy = start.pose.position.y - goal.pose.position.y;
174 
175  double length = sqrt(dx * dx + dy * dy);
176 
177  geometry_msgs::msg::PoseStamped prevState;
178  if (length > skip_straight_motion_distance_)
179  {
180  // skip initial pure spinning and initial straight motion
181  RCLCPP_INFO(
182  nh_->get_logger(), "1 - heading to goal position pure spinning radstep: %lf",
183  puresSpinningRadStep_);
184  double heading_direction = atan2(dy, dx);
185  double startyaw = tf2::getYaw(q);
186  double offset = angles::shortest_angular_distance(startyaw, heading_direction);
187  heading_direction = startyaw + offset;
188 
189  prevState = cl_move_base_z::makePureSpinningSubPlan(
190  start, heading_direction, plan, puresSpinningRadStep_);
191  RCLCPP_INFO(nh_->get_logger(), "2 - going forward keep orientation pure straight");
192 
193  prevState =
194  cl_move_base_z::makePureStraightSubPlan(prevState, goal.pose.position, length, plan);
195  }
196  else
197  {
198  prevState = start;
199  }
200 
201  RCLCPP_WARN_STREAM(
202  nh_->get_logger(), "[BackwardGlobalPlanner] backward global plan size: " << plan.size());
203 }

210 nav_msgs::msg::Path BackwardGlobalPlanner::createPlan(
211  const geometry_msgs::msg::PoseStamped & start, const geometry_msgs::msg::PoseStamped & goal)
212 {
213  RCLCPP_INFO_STREAM(
214  nh_->get_logger(), "[BackwardGlobalPlanner] goal frame id: "
215  << goal.header.frame_id << " pose: " << goal.pose.position);
216  RCLCPP_INFO_STREAM(
217  nh_->get_logger(), "[BackwardGlobalPlanner] goal pose frame id: " << goal.header.frame_id);
218 
219  rclcpp::Duration ttol = rclcpp::Duration::from_seconds(transform_tolerance_);
220  //---------------------------------------------------------------------
221  geometry_msgs::msg::PoseStamped transformedStart;
222  nav_2d_utils::transformPose(tf_, costmap_ros_->getGlobalFrameID(), start, transformedStart, ttol);
223  transformedStart.header.frame_id = costmap_ros_->getGlobalFrameID();
224  //---------------------------------------------------------------------
225  geometry_msgs::msg::PoseStamped transformedGoal;
226  nav_2d_utils::transformPose(tf_, costmap_ros_->getGlobalFrameID(), goal, transformedGoal, ttol);
227  transformedGoal.header.frame_id = costmap_ros_->getGlobalFrameID();
228 
229  RCLCPP_INFO_STREAM(
230  nh_->get_logger(), "[BackwardGlobalPlanner] creating default backward global plan");
231  //---------------------------------------------------------------------
232  std::vector<geometry_msgs::msg::PoseStamped> plan;
233  this->createDefaultBackwardPath(transformedStart, transformedGoal, plan);
234 
235  RCLCPP_INFO_STREAM(nh_->get_logger(), "[BackwardGlobalPlanner] publishing markers");
236  publishGoalMarker(transformedGoal.pose, 1.0, 0, 1.0);
237 
238  nav_msgs::msg::Path planMsg;
239  planMsg.poses = plan;
240  planMsg.header.stamp = this->nh_->now();
241  planMsg.header.frame_id = this->costmap_ros_->getGlobalFrameID();
242 
243  //---------------------------------------------------------------------
244  // check plan rejection if obstacle is found
245  bool acceptedGlobalPlan = true;
246 
247  RCLCPP_INFO_STREAM(
248  nh_->get_logger(), "[BackwardGlobalPlanner] checking backwards trajectory on costmap");
249  auto costmap2d = this->costmap_ros_->getCostmap();
250  for (auto & p : plan)
251  {
252  unsigned int mx, my;
253  costmap2d->worldToMap(p.pose.position.x, p.pose.position.y, mx, my);
254  auto cost = costmap2d->getCost(mx, my);
255 
256  // static const unsigned char NO_INFORMATION = 255;
257  // static const unsigned char LETHAL_OBSTACLE = 254;
258  // static const unsigned char INSCRIBED_INFLATED_OBSTACLE = 253;
259  // static const unsigned char FREE_SPACE = 0;
260  if (cost >= nav2_costmap_2d::INSCRIBED_INFLATED_OBSTACLE)
261  {
262  acceptedGlobalPlan = false;
263  break;
264  }
265  }
266 
267  if (!acceptedGlobalPlan)
268  {
269  RCLCPP_WARN_STREAM(
270  nh_->get_logger(),
271  "[BackwardGlobalPlanner] backward plan request is not accepted, returning "
272  "empty path");
273  planMsg.poses.clear();
274  }
275 
276  RCLCPP_WARN_STREAM(
277  nh_->get_logger(), "[BackwardGlobalPlanner] backward global plan publishing path. poses count: "
278  << planMsg.poses.size());
279  planPub_->publish(planMsg);
280 
281  return planMsg;
282 }
