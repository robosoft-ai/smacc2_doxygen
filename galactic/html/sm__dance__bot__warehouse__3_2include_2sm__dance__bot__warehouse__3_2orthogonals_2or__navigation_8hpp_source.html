84 void loadWaypointsFromYaml(WaypointNavigator * waypointsNavigator)
85 {
86 // if it is the first time and the waypoints navigator is not configured
87 std::string planfilepath;
88 getNode()->declare_parameter("waypoints_plan", planfilepath);
89 if (getNode()->get_parameter("waypoints_plan", planfilepath))
90 {
91 std::string package_share_directory =
92 ament_index_cpp::get_package_share_directory("sm_dance_bot_warehouse_3");
93 boost::replace_all(planfilepath, "$(pkg_share)", package_share_directory);
94 waypointsNavigator->loadWayPointsFromFile(planfilepath);
95 RCLCPP_INFO(getLogger(), "waypoints plan: %s", planfilepath.c_str());
96 }
97 else
98 {
99 RCLCPP_ERROR(getLogger(), "waypoints plan file not found: NONE");
100 }
