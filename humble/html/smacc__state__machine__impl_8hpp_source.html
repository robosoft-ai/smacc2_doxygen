54template <typename TOrthogonal>
55TOrthogonal * ISmaccStateMachine::getOrthogonal()
56{
57 std::lock_guard<std::recursive_mutex> lock(m_mutex_);
58
59 std::string orthogonalkey = demangledTypeName<TOrthogonal>();
60 TOrthogonal * ret;
61
62 auto it = orthogonals_.find(orthogonalkey);
63
64 if (it != orthogonals_.end())
65 {
66 RCLCPP_DEBUG(
67 getLogger(),
68 "Orthogonal %s resource is being required from some state, client or component. Found "
69 "resource in "
70 "cache.",
71 orthogonalkey.c_str());
72 ret = dynamic_cast<TOrthogonal *>(it->second.get());
73 return ret;
74 }
75 else
76 {
77 std::stringstream ss;
78 ss << "Orthogonal not found " << orthogonalkey.c_str() << std::endl;
79 ss << "The existing orthogonals are the following: " << std::endl;
80 for (auto & orthogonal : orthogonals_)
81 {
82 ss << " - " << orthogonal.first << std::endl;
83 }
84
85 RCLCPP_WARN_STREAM(getLogger(), ss.str());
86
87 return nullptr;
88 }
89}

117 {
118 ss << " - " << orthogonal.first << std::endl;

120 RCLCPP_WARN_STREAM(getLogger(), ss.str());
121 }
122 //this->unlockStateMachine("create orthogonal");

177template <typename EventType>
178void ISmaccStateMachine::postEvent(EventType * ev, EventLifeTime evlifetime)
179{
180 std::lock_guard<std::recursive_mutex> guard(eventQueueMutex_);
181
182#define eventtypename demangleSymbol<EventType>().c_str()
183
184 TRACEPOINT(smacc2_event, eventtypename);
185
186 if (
187 evlifetime == EventLifeTime::CURRENT_STATE &&
188 (stateMachineCurrentAction == StateMachineInternalAction::STATE_EXITING ||
189 stateMachineCurrentAction == StateMachineInternalAction::TRANSITIONING))
190 {
191 RCLCPP_WARN_STREAM(
192 getLogger(),
193 "[ISmaccStateMachine] CURRENT STATE SCOPED EVENT DISCARDED, state is exiting/transitioning "
194 << eventtypename);
195 return;
196 // in this case we may lose/skip events, if this is not right for some cases we should create a
197 // queue to lock the events during the transitions. This issues appeared when a client
198 // asyncbehavior was posting an event meanwhile we were doing the transition, but the main
199 // thread was waiting for its correct finalization (with thread.join)
200 }
201
202 // when a postting event is requested by any component, client, or client behavior
203 // we reach this place. Now, we propagate the events to all the state state reactors to generate
204 // some more events
205
206 RCLCPP_DEBUG_STREAM(getLogger(), "[PostEvent entry point] " << eventtypename);
207 for (auto currentState : currentState_)
208 {
209 propagateEventToStateReactors(currentState, ev);
210 }
211
212 this->signalDetector_->postEvent(ev);
213}

217{
218 auto evname = smacc2::introspection::demangleSymbol<EventType>();
219 RCLCPP_INFO_STREAM(getLogger(), "Event " << evname);
220 auto * ev = new EventType();
221 this->postEvent(ev, evlifetime);

224template <typename T>
225bool ISmaccStateMachine::getGlobalSMData(std::string name, T & ret)
226{
227 std::lock_guard<std::recursive_mutex> lock(m_mutex_);
228 // RCLCPP_WARN(getLogger(),"get SM Data lock acquire");
229 bool success = false;
230
231 if (!globalData_.count(name))
232 {
233 // RCLCPP_WARN(getLogger(),"get SM Data - data do not exist");
234 success = false;
235 }
236 else
237 {
238 // RCLCPP_WARN(getLogger(),"get SM DAta -data exist. accessing");
239 try
240 {
241 auto & v = globalData_[name];
242
243 // RCLCPP_WARN(getLogger(),"get SM DAta -data exist. any cast");
244 ret = boost::any_cast<T>(v.second);
245 success = true;
246 // RCLCPP_WARN(getLogger(),"get SM DAta -data exist. success");
247 }
248 catch (boost::bad_any_cast & ex)
249 {
250 RCLCPP_ERROR(getLogger(), "bad any cast: %s", ex.what());
251 success = false;
252 }
253 }
254
255 // RCLCPP_WARN(getLogger(),"get SM Data lock release");
256 return success;
257}

280template <typename StateField, typename BehaviorType>
281void ISmaccStateMachine::mapBehavior()
282{
283 std::string stateFieldName = demangleSymbol(typeid(StateField).name());
284 std::string behaviorType = demangleSymbol(typeid(BehaviorType).name());
285 RCLCPP_INFO(
286 getLogger(), "Mapping state field '%s' to stateReactor '%s'", stateFieldName.c_str(),
287 behaviorType.c_str());
288 SmaccClientBehavior * globalreference;
289 if (!this->getGlobalSMData(stateFieldName, globalreference))
290 {
291 // Using the requires component approach, we force a unique existence
292 // of this component
293 BehaviorType * behavior;
294 this->requiresComponent(behavior);
295 globalreference = dynamic_cast<ISmaccClientBehavior *>(behavior);
296
297 this->setGlobalSMData(stateFieldName, globalreference);
298 }
299}

321

411template <typename TSmaccSignal, typename TMemberFunctionPrototype, typename TSmaccObjectType>
412boost::signals2::connection ISmaccStateMachine::createSignalConnection(
413 TSmaccSignal & signal, TMemberFunctionPrototype callback, TSmaccObjectType * object)
414{
415 std::lock_guard<std::recursive_mutex> lock(m_mutex_);
416
417 static_assert(
418 std::is_base_of<ISmaccState, TSmaccObjectType>::value ||
419 std::is_base_of<ISmaccClient, TSmaccObjectType>::value ||
420 std::is_base_of<ISmaccClientBehavior, TSmaccObjectType>::value ||
421 std::is_base_of<StateReactor, TSmaccObjectType>::value ||
422 std::is_base_of<ISmaccComponent, TSmaccObjectType>::value,
423 "Only are accepted smacc types as subscribers for smacc signals");
424
425 typedef decltype(callback) ft;
426 Bind<boost::function_types::function_arity<ft>::value> binder;
427 boost::signals2::connection connection;
428
429 // long life-time objects
430 if (
431 std::is_base_of<ISmaccComponent, TSmaccObjectType>::value ||
432 std::is_base_of<ISmaccClient, TSmaccObjectType>::value ||
433 std::is_base_of<ISmaccOrthogonal, TSmaccObjectType>::value ||
434 std::is_base_of<ISmaccStateMachine, TSmaccObjectType>::value)
435 {
436 connection = binder.bindaux(signal, callback, object, nullptr);
437 }
438 else if (
439 std::is_base_of<ISmaccState, TSmaccObjectType>::value ||
440 std::is_base_of<StateReactor, TSmaccObjectType>::value ||
441 std::is_base_of<ISmaccClientBehavior, TSmaccObjectType>::value)
442 {
443 RCLCPP_INFO(
444 getLogger(),
445 "[StateMachine] life-time constrained smacc signal subscription created. Subscriber is %s",
446 demangledTypeName<TSmaccObjectType>().c_str());
447
448 std::shared_ptr<CallbackCounterSemaphore> callbackCounterSemaphore;
449 if (stateCallbackConnections.count(object))
450 {
451 callbackCounterSemaphore = stateCallbackConnections[object];
452 }
453 else
454 {
455 callbackCounterSemaphore =
456 std::make_shared<CallbackCounterSemaphore>(demangledTypeName<TSmaccObjectType>().c_str());
457 stateCallbackConnections[object] = callbackCounterSemaphore;
458 }
459
460 connection = binder.bindaux(signal, callback, object, callbackCounterSemaphore);
461 callbackCounterSemaphore->addConnection(connection);
462 }
463 else // state life-time objects
464 {
465 RCLCPP_WARN(
466 getLogger(),
467 "[StateMachine] connecting signal to an unknown object with life-time unknown "
468 "behavior. It might provoke "
469 "an exception if the object is destroyed during the execution.");
470
471 connection = binder.bindaux(signal, callback, object, nullptr);
472 }
473
474 return connection;
475}

493template <typename StateType>
494void ISmaccStateMachine::notifyOnStateEntryEnd(StateType *)
495{
496 RCLCPP_INFO(
497 getLogger(), "[%s] State OnEntry code finished",
498 demangleSymbol(typeid(StateType).name()).c_str());
499
500 auto currentState = this->currentState_.back();
501 for (auto pair : this->orthogonals_)
502 {
503 RCLCPP_DEBUG(getLogger(), "ortho onentry: %s", pair.second->getName().c_str());
504 auto & orthogonal = pair.second;
505 try
506 {
507 orthogonal->onEntry();
508 }
509 catch (const std::exception & e)
510 {
511 RCLCPP_ERROR(
512 getLogger(),
513 "[Orthogonal %s] Exception on Entry - continuing with next orthogonal. Exception info: %s",
514 pair.second->getName().c_str(), e.what());
515 }
516 }
517
518 for (auto & sr : currentState->getStateReactors())
519 {
520 auto srname = smacc2::demangleSymbol(typeid(*sr).name());
521 RCLCPP_INFO_STREAM(getLogger(), "state reactor onEntry: " << srname);
522 try
523 {
524 sr->onEntry();
525 }
526 catch (const std::exception & e)
527 {
528 RCLCPP_ERROR(
529 getLogger(),
530 "[State Reactor %s] Exception on Entry - continuing with next state reactor. Exception "
531 "info: %s",
532 srname.c_str(), e.what());
533 }
534 }
535
536 for (auto & eg : currentState->getEventGenerators())
537 {
538 auto egname = smacc2::demangleSymbol(typeid(*eg).name());
539 RCLCPP_INFO_STREAM(getLogger(), "event generator onEntry: " << egname);
540 try
541 {
542 eg->onEntry();
543 }
544 catch (const std::exception & e)
545 {
546 RCLCPP_ERROR(
547 getLogger(),
548 "[Event generator %s] Exception on Entry - continuing with next state reactor. Exception "
549 "info: %s",
550 egname.c_str(), e.what());
551 }
552 }
553
554 {
555 std::lock_guard<std::recursive_mutex> lock(m_mutex_);
556 this->updateStatusMessage();
557 stateMachineCurrentAction = StateMachineInternalAction::STATE_RUNNING;
558 }
559}

648template <typename StateType>
649void ISmaccStateMachine::notifyOnStateExited(StateType * state)
650{
651 this->lockStateMachine("state exit");
652
653 signalDetector_->notifyStateExited(state);
654
655 auto fullname = demangleSymbol(typeid(StateType).name());
656 RCLCPP_WARN_STREAM(getLogger(), "exiting state: " << fullname);
657
658 RCLCPP_INFO_STREAM(getLogger(), "Notification State Disposing: leaving state" << state);
659 for (auto pair : this->orthogonals_)
660 {
661 auto & orthogonal = pair.second;
662 try
663 {
664 orthogonal->onDispose();
665 }
666 catch (const std::exception & e)
667 {
668 RCLCPP_ERROR(
669 getLogger(),
670 "[Orthogonal %s] Exception onDispose - continuing with next orthogonal. Exception info: %s",
671 pair.second->getName().c_str(), e.what());
672 }
673 }
674
675 for (auto & sr : state->getStateReactors())
676 {
677 auto srname = smacc2::demangleSymbol(typeid(*sr).name()).c_str();
678 RCLCPP_INFO(getLogger(), "state reactor disposing: %s", srname);
679 try
680 {
681 this->disconnectSmaccSignalObject(sr.get());
682 }
683 catch (const std::exception & e)
684 {
685 RCLCPP_ERROR(
686 getLogger(),
687 "[State Reactor %s] Exception on OnDispose - continuing with next state reactor. Exception "
688 "info: %s",
689 srname, e.what());
690 }
691 }
692
693 for (auto & eg : state->getEventGenerators())
694 {
695 auto egname = smacc2::demangleSymbol(typeid(*eg).name()).c_str();
696 RCLCPP_INFO(getLogger(), "state reactor disposing: %s", egname);
697 try
698 {
699 this->disconnectSmaccSignalObject(eg.get());
700 }
701 catch (const std::exception & e)
702 {
703 RCLCPP_ERROR(
704 getLogger(),
705 "[State Reactor %s] Exception on OnDispose - continuing with next state reactor. Exception "
706 "info: %s",
707 egname, e.what());
708 }
709 }
710
711 this->stateCallbackConnections.clear();
712 currentState_.pop_back();
713
714 // then call exit state
715 RCLCPP_WARN_STREAM(getLogger(), "state exit: " << fullname);
716
717 stateMachineCurrentAction = StateMachineInternalAction::TRANSITIONING;
718 this->unlockStateMachine("state exit");
719}

721template <typename EventType>
722void ISmaccStateMachine::propagateEventToStateReactors(ISmaccState * st, EventType * ev)
723{
724 RCLCPP_DEBUG(
725 getLogger(), "PROPAGATING EVENT [%s] TO SRs [%s]: ", demangleSymbol<EventType>().c_str(),
726 st->getClassName().c_str());
727 for (auto & sb : st->getStateReactors())
728 {
729 sb->notifyEvent(ev);
730 }
731
732 auto * pst = st->getParentState();
733 if (pst != nullptr)
734 {
735 propagateEventToStateReactors(pst, ev);
736 }
737}
